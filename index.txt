        function playSynthSound(freqs, duration, type = 'sine', delayBetween = 0) {
            if (audioCtx.state === 'suspended') audioCtx.resume();
            freqs.forEach((freq, index) => {
                setTimeout(() => {
                    const osc = audioCtx.createOscillator();
                    const gain = audioCtx.createGain();
                    osc.type = type;
                    osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
                    gain.gain.setValueAtTime(0.12, audioCtx.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
                    osc.connect(gain); gain.connect(audioCtx.destination);
                    osc.start(); osc.stop(audioCtx.currentTime + duration);
                }, index * delayBetween * 1000);
            });
        }

        function nextWord() { playClickSFX(); isCardFlipped = false; document.getElementById('flashcard').classList.remove('rotate-y-180'); setTimeout(() => { flashcardIndex = (flashcardIndex + 1) % categories[currentCategory].length; updateFlashcard(); }, 150); }
        function prevWord() { playClickSFX(); isCardFlipped = false; document.getElementById('flashcard').classList.remove('rotate-y-180'); setTimeout(() => { flashcardIndex = (flashcardIndex - 1 + categories[currentCategory].length) % categories[currentCategory].length; updateFlashcard(); }, 150); }

        function speak(t) { if ('speechSynthesis' in window) { window.speechSynthesis.cancel(); const u = new SpeechSynthesisUtterance(t); u.lang = document.getElementById('accent-selector').value; u.rate = parseFloat(document.getElementById('speed-selector').value); window.speechSynthesis.speak(u); } }

        function initQuiz() {
            const list = categories[currentCategory];
            if (quizIndex >= list.length) { celebrate(); return; }
            const cur = list[quizIndex];
            document.getElementById('quiz-question-word').innerText = cur.word;
            document.getElementById('quiz-score').innerText = `Score: ${quizScore}`;
            document.getElementById('quiz-streak').innerText = `Streak: 🔥 ${quizStreak}`;
            let opts = [cur, ...list.filter(i=>i.word!==cur.word).sort(()=>0.5-Math.random()).slice(0,3)].sort(()=>0.5-Math.random());
            const container = document.getElementById('options-container');
            container.innerHTML = '';
            opts.forEach(o => {
                const b = document.createElement('button');
                b.className = 'bg-white border-2 border-slate-200 rounded-2xl p-4 hover:border-slate-400 active:scale-95 transition-all shadow-sm flex flex-col items-center gap-2';
                b.onclick = () => check(o, b);
                b.innerHTML = `<div class="w-24 h-24 rounded-xl overflow-hidden shadow-inner"><img class="w-full h-full object-cover" src="${o.img}"></div><span class="font-bold text-slate-700">${o.spanish}</span>`;
                container.appendChild(b);
            });
            document.getElementById('quiz-progress-bar').style.width = `${((quizIndex+1)/list.length)*100}%`;
            document.getElementById('quiz-counter').innerText = `Question ${quizIndex+1} of ${list.length}`;
            setTimeout(() => speak(cur.word), 350);
        }

        function speakQuizWord() { speak(categories[currentCategory][quizIndex].word); }
        function check(o, b) {
            const cur = categories[currentCategory][quizIndex];
            document.querySelectorAll('#options-container button').forEach(btn => btn.style.pointerEvents = 'none');
            if (o.word === cur.word) { playSuccessSFX(); b.classList.add('border-emerald-500', 'bg-emerald-50'); quizScore += 10 + (quizStreak * 2); quizStreak++; }
            else { playErrorSFX(); b.classList.add('border-rose-500', 'bg-rose-50'); quizStreak = 0; }
            setTimeout(() => { quizIndex++; initQuiz(); }, 1500);
        }

        function celebrate() { playSuccessSFX(); confettiArr = []; for(let i=0; i<150; i++) confettiArr.push(new Confetto()); cancelAnimationFrame(animId); animConfetti(); document.getElementById('modal-score').innerText = `${quizScore} pts`; document.getElementById('modal-message').innerText = `Unit "${currentCategory.toUpperCase()}" finished!`; document.getElementById('celebration-modal').classList.remove('hidden'); setTimeout(()=>document.getElementById('celebration-modal').classList.add('opacity-100'),50); }
        function closeModal() { document.getElementById('celebration-modal').classList.remove('opacity-100'); setTimeout(()=>{ document.getElementById('celebration-modal').classList.add('hidden'); switchMode('flashcard'); },300); }

        const canvas = document.getElementById('confetti-canvas'), ctx = canvas.getContext('2d');
        let confettiArr = [], animId;
        function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        window.addEventListener('resize', resize); resize();
        class Confetto { constructor() { this.x = Math.random()*canvas.width; this.y = Math.random()*canvas.height-canvas.height; this.r = Math.random()*6+4; this.d = Math.random()*canvas.height; this.color = `hsl(${Math.random()*360},90%,65%)`; this.tilt = Math.random()*10-5; this.tiltAngleIncremental = Math.random()*0.07+0.02; this.tiltAngle = 0; } draw() { ctx.beginPath(); ctx.lineWidth = this.r/2; ctx.strokeStyle = this.color; ctx.moveTo(this.x+this.tilt+this.r/4, this.y); ctx.lineTo(this.x+this.tilt, this.y+this.tilt+this.r/4); ctx.stroke(); } }
        function animConfetti() { ctx.clearRect(0,0,canvas.width,canvas.height); let active = false; confettiArr.forEach(c => { c.tiltAngle += c.tiltAngleIncremental; c.y += (Math.cos(c.d)+3+c.r/2)/2; c.x += Math.sin(c.tiltAngle); c.tilt = Math.sin(c.tiltAngle-c.r/2)*15; if(c.y < canvas.height) active = true; c.draw(); }); if(active) animId = requestAnimationFrame(animConfetti); }

        window.onload = () => { updateFlashcard(); applyTheme(); };
